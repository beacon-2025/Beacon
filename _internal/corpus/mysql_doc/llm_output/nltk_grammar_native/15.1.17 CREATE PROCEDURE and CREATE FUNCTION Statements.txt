```
S -> CREATE_PROCEDURE_STATEMENT | CREATE_FUNCTION_STATEMENT

CREATE_PROCEDURE_STATEMENT -> 'CREATE' DEFINER_OPTION 'PROCEDURE' IF_NOT_EXISTS_OPTION 'new_sp_name_1' '(' PROC_PARAMETER_LIST ')' CHARACTERISTIC_LIST ROUTINE_BODY
CREATE_FUNCTION_STATEMENT -> 'CREATE' DEFINER_OPTION 'FUNCTION' IF_NOT_EXISTS_OPTION 'new_sp_name_1' '(' FUNC_PARAMETER_LIST ')' 'RETURNS' TYPE CHARACTERISTIC_LIST ROUTINE_BODY

DEFINER_OPTION -> 'DEFINER' '=' 'user_name_1' | ''
IF_NOT_EXISTS_OPTION -> 'IF NOT EXISTS' | ''

PROC_PARAMETER_LIST -> PROC_PARAMETER | PROC_PARAMETER ',' PROC_PARAMETER_LIST | ''
PROC_PARAMETER -> MODE 'param_name_1' TYPE
MODE -> 'IN' | 'OUT' | 'INOUT'

FUNC_PARAMETER_LIST -> FUNC_PARAMETER | FUNC_PARAMETER ',' FUNC_PARAMETER_LIST | ''
FUNC_PARAMETER -> 'param_name_1' TYPE

TYPE -> 'valid_data_type'
CHARACTERISTIC_LIST -> CHARACTERISTIC | CHARACTERISTIC CHARACTERISTIC_LIST | ''
CHARACTERISTIC -> COMMENT_OPTION | LANGUAGE_OPTION | DETERMINISTIC_OPTION 
                | SQL_DATA_OPTION | SQL_SECURITY_OPTION

COMMENT_OPTION -> 'COMMENT' "'string_value'"
LANGUAGE_OPTION -> 'LANGUAGE' 'SQL' | 'LANGUAGE' 'JAVASCRIPT'
DETERMINISTIC_OPTION -> 'DETERMINISTIC' | 'NOT DETERMINISTIC'
SQL_DATA_OPTION -> 'CONTAINS SQL' | 'NO SQL' | 'READS SQL DATA' | 'MODIFIES SQL DATA'
SQL_SECURITY_OPTION -> 'SQL SECURITY' 'DEFINER' | 'SQL SECURITY' 'INVOKER'

ROUTINE_BODY -> 'routine_body_unknown'
```
