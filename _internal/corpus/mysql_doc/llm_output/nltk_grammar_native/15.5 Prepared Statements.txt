S -> PREPARE_STATEMENT | EXECUTE_STATEMENT | DEALLOCATE_PREPARE_STATEMENT

PREPARE_STATEMENT -> 'PREPARE' STATEMENT_NAME FROM STATEMENT_TEXT
EXECUTE_STATEMENT -> 'EXECUTE' STATEMENT_NAME USING PARAMETERS
DEALLOCATE_PREPARE_STATEMENT -> 'DEALLOCATE PREPARE' STATEMENT_NAME

STATEMENT_NAME -> 'statement_name_1'
STATEMENT_TEXT -> 'statement_text_unknown'  # Can be a string literal or user variable

PARAMETERS -> PARAMETER | PARAMETER ',' PARAMETERS
PARAMETER -> 'parameter_unknown'  # Represents using placeholders

ALLOWED_SQL_STATEMENTS -> ALTER_STATEMENT | ANALYZE_STATEMENT | CALL_STATEMENT | CHANGE_STATEMENT |
                           CHECKSUM_STATEMENT | COMMIT_STATEMENT | CREATE_DROP_RENAME_STATEMENT |
                           DEALLOCATE_STATEMENT | DROP_VIEW_STATEMENT | DELETE_STATEMENT | DO_STATEMENT |
                           EXECUTE_STATEMENT | FLUSH_STATEMENT | GRANT_STATEMENT | INSERT_STATEMENT |
                           INSTALL_STATEMENT | KILL_STATEMENT | OPTIMIZE_STATEMENT | PREPARE_STATEMENT |
                           REPAIR_STATEMENT | REPLACE_STATEMENT | REPLICA_STATEMENT | RESET_STATEMENT |
                           REVOKE_STATEMENT | SELECT_STATEMENT | SET_STATEMENT | SHOW_STATEMENT |
                           SHOW_CREATE_STATEMENT | TRUNCATE_STATEMENT | UNINSTALL_PLUGIN_STATEMENT | 
                           UPDATE_STATEMENT

ALTER_STATEMENT -> 'ALTER' TABLE_USER
TABLE_USER -> 'INSTANCE' | 'TABLE' | 'USER'

CHANGE_STATEMENT -> 'CHANGE' REPLICATION
REPLICATION -> 'REPLICATION SOURCE TO' | 'REPLICATION FILTER'

GRANT_STATEMENT -> 'GRANT' ROLE_GRANT
ROLE_GRANT -> 'ROLE' | ''

CREATE_DROP_RENAME_STATEMENT -> 'CREATE' OBJECT | 'DROP' OBJECT | 'RENAME' OBJECT

OBJECT -> 'INDEX' | 'DATABASE' | 'TABLE' | 'USER'

SHOW_STATEMENT -> 'SHOW' SHOW_OPTIONS
SHOW_OPTIONS -> 'BINLOG EVENTS' | 'BINARY LOGS' | 'BINARY LOG STATUS' | 'CHARACTER SETS' | 'COLLATIONS' | 'DATABASES' |
                'ENGINES' | 'ERRORS' | 'EVENTS' | 'FIELDS' | 'FUNCTION CODE' | 'FUNCTION STATUS' | 'GRANTS' | 'KEYS' |
                'OPEN TABLES' | 'PLUGINS' | 'PRIVILEGES' | 'PROCEDURE CODE' | 'PROCEDURE STATUS' | 'PROCESSLIST' |
                'PROFILE' | 'PROFILES' | 'RELAYLOG EVENTS' | 'REPLICAS' | 'REPLICA STATUS' | 'STATUS' |
                'PROCEDURE STATUS' | 'TABLE STATUS' | 'TABLES' | 'TRIGGERS' | 'VARIABLES' | 'WARNINGS'

SHOW_CREATE_STATEMENT -> 'SHOW CREATE' SHOW_CREATE_OPTIONS
SHOW_CREATE_OPTIONS -> 'DATABASE' | 'EVENT' | 'FUNCTION' | 'PROCEDURE' | 'TABLE' | 'TRIGGER' | 'USER' | 'VIEW'

DEALLOCATE_STATEMENT -> 'DEALLOCATE PREPARE'
REPLICA_STATEMENT -> 'REPLICA' REPLICA_ACTION
REPLICA_ACTION -> 'START' | 'STOP'

REVOKE_STATEMENT -> 'REVOKE' REVOKE_OPTIONS
REVOKE_OPTIONS -> 'ALL' | 'ROLE'
