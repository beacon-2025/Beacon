```
S -> DECLARE_CONDITION_STATEMENT | DECLARE_HANDLER_STATEMENT | SIGNAL_STATEMENT | RESIGNAL_STATEMENT | GET_DIAGNOSTICS_STATEMENT

DECLARE_CONDITION_STATEMENT -> 'DECLARE' 'condition_name' 'CONDITION FOR' CONDITION_VALUE
CONDITION_VALUE -> MYSQL_ERROR_CODE | SQLSTATE_VALUE
MYSQL_ERROR_CODE -> "'mysql_error_code_value'"
SQLSTATE_VALUE -> 'SQLSTATE' 'VALUE' "'sqlstate_value'"

DECLARE_HANDLER_STATEMENT -> 'DECLARE' HANDLER_ACTION 'HANDLER' 'FOR' CONDITION_VALUE_LIST DECLARE_STATEMENT
HANDLER_ACTION -> 'CONTINUE' | 'EXIT' | 'UNDO'
CONDITION_VALUE_LIST -> CONDITION_VALUE | CONDITION_VALUE ',' CONDITION_VALUE_LIST

SIGNAL_STATEMENT -> 'SIGNAL' CONDITION_VALUE SIGNAL_INFORMATION
SIGNAL_INFORMATION -> 'SET' SIGNAL_INFORMATION_ITEM | ''
SIGNAL_INFORMATION_ITEM -> CONDITION_INFORMATION_ITEM_NAME '=' SIMPLE_VALUE_SPECIFICATION
CONDITION_INFORMATION_ITEM_NAME -> 'CLASS_ORIGIN' | 'SUBCLASS_ORIGIN' | 'MESSAGE_TEXT' 
                                 | 'MYSQL_ERRNO' | 'CONSTRAINT_CATALOG' | 'CONSTRAINT_SCHEMA'
                                 | 'CONSTRAINT_NAME' | 'CATALOG_NAME' | 'SCHEMA_NAME' 
                                 | 'TABLE_NAME' | 'COLUMN_NAME' | 'CURSOR_NAME'
SIMPLE_VALUE_SPECIFICATION -> "'simple_value_specification_unknown'"

RESIGNAL_STATEMENT -> 'RESIGNAL' RESIGNAL_CONDITION_VALUE SIGNAL_INFORMATION
RESIGNAL_CONDITION_VALUE -> CONDITION_VALUE | ''

GET_DIAGNOSTICS_STATEMENT -> 'GET' DIAGNOSTICS_TYPE DIAGNOSTICS_INFO
DIAGNOSTICS_TYPE -> 'CURRENT' | 'STACKED' | ''
DIAGNOSTICS_INFO -> STATEMENT_INFORMATION | CONDITION_INFORMATION
STATEMENT_INFORMATION -> STATEMENT_INFORMATION_ITEM
CONDITION_INFORMATION -> CONDITION_NUMBER CONDITION_INFORMATION_ITEM
STATEMENT_INFORMATION_ITEM -> 'NUMBER' | 'ROW_COUNT'
CONDITION_INFORMATION_ITEM -> CONDITION_INFORMATION_ITEM_NAME '=' CONDITION_ITEM_TARGET
CONDITION_ITEM_TARGET -> "'condition_item_target_unknown'"
CONDITION_NUMBER -> "'condition_number_unknown'"
```
