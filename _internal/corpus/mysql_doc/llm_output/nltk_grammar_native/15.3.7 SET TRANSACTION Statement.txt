```
S -> SET_TRANSACTION_STATEMENT
SET_TRANSACTION_STATEMENT -> 'SET' SCOPE 'TRANSACTION' TRANSACTION_CHARACTERISTICS

SCOPE -> 'GLOBAL' | 'SESSION' | ''

TRANSACTION_CHARACTERISTICS -> TRANSACTION_CHARACTERISTIC | TRANSACTION_CHARACTERISTIC ',' TRANSACTION_CHARACTERISTICS

TRANSACTION_CHARACTERISTIC -> ISOLATION_LEVEL | ACCESS_MODE
ISOLATION_LEVEL -> 'ISOLATION LEVEL' LEVEL
LEVEL -> 'REPEATABLE READ' | 'READ COMMITTED' | 'READ UNCOMMITTED' | 'SERIALIZABLE'

ACCESS_MODE -> 'READ WRITE' | 'READ ONLY'
```
