S -> LOCK_TABLE_STATEMENT | UNLOCK_TABLE_STATEMENT

LOCK_TABLE_STATEMENT -> 'LOCK' TABLE_CLAUSE TABLE_LIST
TABLE_CLAUSE -> 'TABLE' | 'TABLES'
TABLE_LIST -> TABLE_SPEC | TABLE_SPEC ',' TABLE_LIST
TABLE_SPEC -> 'table_name_1' ALIAS LOCK_TYPE
ALIAS -> 'AS' 'alias_unknown' | '' 

LOCK_TYPE -> 'READ' LOCAL_OPTION | 'WRITE'
LOCAL_OPTION -> 'LOCAL' | ''

UNLOCK_TABLE_STATEMENT -> 'UNLOCK' TABLE_CLAUSE
