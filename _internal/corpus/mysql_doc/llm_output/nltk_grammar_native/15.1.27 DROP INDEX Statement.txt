```
S -> DROP_INDEX_STATEMENT
DROP_INDEX_STATEMENT -> 'DROP' 'INDEX' 'index_name_1' 'ON' 'table_name_1' OPTIONS
OPTIONS -> OPTION | OPTION OPTIONS
OPTION -> ALGORITHM_OPTION | LOCK_OPTION

ALGORITHM_OPTION -> 'ALGORITHM' '=' ALGORITHM_VALUE
ALGORITHM_VALUE -> "'DEFAULT'" | "'INPLACE'" | "'COPY'"
LOCK_OPTION -> 'LOCK' '=' LOCK_VALUE
LOCK_VALUE -> "'DEFAULT'" | "'NONE'" | "'SHARED'" | "'EXCLUSIVE'"
```
