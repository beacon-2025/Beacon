```
S -> LOAD_DATA_STATEMENT

LOAD_DATA_STATEMENT -> 'LOAD' 'DATA' LOAD_PRIORITY LOCAL 'INFILE' FILE_NAME LOAD_OPTION 
                        'INTO TABLE' TABLE_NAME PARTITION CHARACTER_SET FIELD_OR_COLUMN LINES IGNORE COLUMNS SETLIST

LOAD_PRIORITY -> 'LOW_PRIORITY' | 'CONCURRENT' | ''
LOCAL -> 'LOCAL' | ''
FILE_NAME -> "'file_name_value'"
LOAD_OPTION -> REPLACE_IGNORE | ''
REPLACE_IGNORE -> 'REPLACE' | 'IGNORE'
TABLE_NAME -> "'table_name_1'"
PARTITION -> 'PARTITION' '(' PARTITION_LIST ')' | ''
PARTITION_LIST -> 'partition_name_1' | 'partition_name_1' ',' PARTITION_LIST
CHARACTER_SET -> 'CHARACTER SET' 'charset_value' | ''
FIELD_OR_COLUMN -> 'FIELDS' FIELD_SPEC | 'COLUMNS' FIELD_SPEC | ''
FIELD_SPEC -> TERMINATED ENCLOSED ESCAPED | ''
TERMINATED -> 'TERMINATED BY' TERMINATED_VALUE | ''
ENCLOSED -> 'OPTIONALLY ENCLOSED BY' ENCLOSED_VALUE | 'ENCLOSED BY' ENCLOSED_VALUE | ''
ESCAPED -> 'ESCAPED BY' ESCAPED_VALUE | ''
TERMINATED_VALUE -> "'string_value'"
ENCLOSED_VALUE -> "'char_value'"
ESCAPED_VALUE -> "'char_value'"
LINES -> 'LINES' LINE_TERMINATED LINE_STARTED | ''
LINE_TERMINATED -> 'TERMINATED BY' TERMINATED_VALUE | ''
LINE_STARTED -> 'STARTING BY' TERMINATED_VALUE | ''
IGNORE -> 'IGNORE' 'number_value' IGNORE_TYPE | ''
IGNORE_TYPE -> 'LINES' | 'ROWS'
COLUMNS -> '(' COL_NAME_OR_VAR_LIST ')' | ''
COL_NAME_OR_VAR_LIST -> 'col_name_or_user_var_1' | 'col_name_or_user_var_1' ',' COL_NAME_OR_VAR_LIST
SETLIST -> SET_STATEMENTS | ''
SET_STATEMENTS -> 'SET' SET_EXPRS | ''
SET_EXPRS -> SET_EXPR | SET_EXPR ',' SET_EXPRS
SET_EXPR -> 'col_name' '=' 'col_expr'
```
