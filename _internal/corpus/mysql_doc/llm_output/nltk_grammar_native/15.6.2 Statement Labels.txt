```
S -> LABEL_BEGIN_BLOCKS | LABEL_LOOP_STATEMENT | LABEL_REPEAT_STATEMENT | LABEL_WHILE_STATEMENT

LABEL_BEGIN_BLOCKS -> BEGIN_LABEL 'BEGIN' STATEMENT_LIST 'END' END_LABEL
LABEL_LOOP_STATEMENT -> BEGIN_LABEL 'LOOP' STATEMENT_LIST 'END LOOP' END_LABEL
LABEL_REPEAT_STATEMENT -> BEGIN_LABEL 'REPEAT' STATEMENT_LIST 'UNTIL' SEARCH_CONDITION 'END REPEAT' END_LABEL
LABEL_WHILE_STATEMENT -> BEGIN_LABEL 'WHILE' SEARCH_CONDITION 'DO' STATEMENT_LIST 'END WHILE' END_LABEL

BEGIN_LABEL -> '[begin_label]' ':'
END_LABEL -> '[end_label]' | ''  # Optional end_labels must match begin_labels if present
STATEMENT_LIST -> 'statement_list' | ''  # Optional statement list
SEARCH_CONDITION -> 'search_condition'

# Rule for using labels with ITERATE and LEAVE
ITERATE_STATEMENT -> 'ITERATE' LABEL
LEAVE_STATEMENT -> 'LEAVE' LABEL
LABEL -> 'label_name_1'

# Example of use in stored procedure
CREATE_PROCEDURE -> 'CREATE' 'PROCEDURE' PROCEDURE_NAME 'BEGIN' LOOP_LABEL STATEMENT_LIST 'END LOOP' LOOP_LABEL 'END'
PROCEDURE_NAME -> 'procedure_name_1'
LOOP_LABEL -> 'label1' ':'
```
