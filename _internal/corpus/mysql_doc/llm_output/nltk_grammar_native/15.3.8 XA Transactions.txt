```
S -> XA_TRANSACTION_STATEMENT

XA_TRANSACTION_STATEMENT -> 'XA' XA_COMMAND XID XID_OPTIONS
XA_COMMAND -> 'START' | 'BEGIN' | 'END' | 'PREPARE' | 'COMMIT' XA_COMMIT_OPTION | 'ROLLBACK' | 'RECOVER' XA_RECOVER_OPTION
XA_COMMIT_OPTION -> 'ONE PHASE' | ''
XA_RECOVER_OPTION -> 'CONVERT XID' | ''
XID -> 'xid_value'

XID_OPTIONS -> JOIN_RESUME_OPTION END_OPTION | ''
JOIN_RESUME_OPTION -> 'JOIN' | 'RESUME' | ''
END_OPTION -> 'SUSPEND' 'FOR MIGRATE' | ''

XID_VALUE_FORMAT -> GTRID ',' BQUAL ',' FORMATID | GTRID ',' BQUAL | GTRID
GTRID -> 'gtrid_value'
BQUAL -> 'bqual_value'
FORMATID -> 'formatID_value'

GTRID -> XA_STRING_LITERAL
BQUAL -> XA_STRING_LITERAL
FORMATID -> 'unsigned_integer_value'

XA_STRING_LITERAL -> QUOTED_STRING | HEX_STRING | BIT_VALUE
QUOTED_STRING -> "'some_string'"
HEX_STRING -> "X'hex_digits'" | "0xhex_digits"
BIT_VALUE -> "b'binary_digits'"
```
