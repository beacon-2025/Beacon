```
S -> ALTER_DATABASE_STATEMENT
ALTER_DATABASE_STATEMENT -> 'ALTER' DATABASE_SCHEMA DATABASE_NAME ALTER_OPTIONS
DATABASE_SCHEMA -> 'DATABASE' | 'SCHEMA'
DATABASE_NAME -> 'db_name_1' | ''  # Empty string if omitted (applies to default database)

ALTER_OPTIONS -> ALTER_OPTION | ALTER_OPTION ALTER_OPTIONS
ALTER_OPTION -> CHARACTER_SET_OPTION | COLLATE_OPTION | ENCRYPTION_OPTION | READ_ONLY_OPTION

CHARACTER_SET_OPTION -> 'DEFAULT' 'CHARACTER SET' '=' 'charset_value'
COLLATE_OPTION -> 'DEFAULT' 'COLLATE' '=' 'collation_value'
ENCRYPTION_OPTION -> 'DEFAULT' 'ENCRYPTION' '=' ENCRYPTION_VALUE
ENCRYPTION_VALUE -> "'Y'" | "'N'"
READ_ONLY_OPTION -> 'READ ONLY' '=' READ_ONLY_VALUE
READ_ONLY_VALUE -> "'DEFAULT'" | "'0'" | "'1'"
```
