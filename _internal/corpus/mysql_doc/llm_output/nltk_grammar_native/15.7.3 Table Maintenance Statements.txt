S -> ANALYZE_TABLE_STATEMENT | ANALYZE_TABLE_HISTOGRAM | DROP_HISTOGRAM | CHECK_TABLE_STATEMENT | CHECKSUM_TABLE_STATEMENT | OPTIMIZE_TABLE_STATEMENT | REPAIR_TABLE_STATEMENT

ANALYZE_OPTION -> 'NO_WRITE_TO_BINLOG' | 'LOCAL' | ''
TABLE_LIST -> TABLE_NAME | TABLE_NAME ',' TABLE_LIST
TABLE_NAME -> 'table_name_1'

ANALYZE_TABLE_HISTOGRAM -> 'ANALYZE' ANALYZE_OPTION 'TABLE' TABLE_NAME 'UPDATE HISTOGRAM ON' COLUMN_LIST HISTOGRAM_BUCKETS UPDATE_OPTION
COLUMN_LIST -> COLUMN_NAME | COLUMN_NAME ',' COLUMN_LIST
COLUMN_NAME -> 'column_name_unknown'
HISTOGRAM_BUCKETS -> 'WITH' NUMERIC_LITERAL 'BUCKETS' | ''
NUMERIC_LITERAL -> 'N_value'
UPDATE_OPTION -> 'MANUAL UPDATE' | 'AUTO UPDATE' | ''

DROP_HISTOGRAM -> 'ANALYZE' ANALYZE_OPTION 'TABLE' TABLE_NAME 'DROP HISTOGRAM ON' COLUMN_LIST

CHECK_TABLE_STATEMENT -> 'CHECK TABLE' TABLE_LIST CHECK_OPTION_LIST
CHECK_OPTION_LIST -> CHECK_OPTION | CHECK_OPTION CHECK_OPTION_LIST
CHECK_OPTION -> 'FOR UPGRADE' | 'QUICK' | 'FAST' | 'MEDIUM' | 'EXTENDED' | 'CHANGED' | ''

CHECKSUM_TABLE_STATEMENT -> 'CHECKSUM TABLE' TABLE_LIST CHECKSUM_OPTION
CHECKSUM_OPTION -> 'QUICK' | 'EXTENDED' | ''

OPTIMIZE_TABLE_STATEMENT -> 'OPTIMIZE' OPTIMIZE_OPTION 'TABLE' TABLE_LIST
OPTIMIZE_OPTION -> 'NO_WRITE_TO_BINLOG' | 'LOCAL' | ''

REPAIR_TABLE_STATEMENT -> 'REPAIR' REPAIR_OPTION 'TABLE' TABLE_LIST REPAIR_OPTIONS
REPAIR_OPTION -> 'NO_WRITE_TO_BINLOG' | 'LOCAL' | ''
REPAIR_OPTIONS -> 'QUICK' | 'EXTENDED' | 'USE_FRM' | ''
