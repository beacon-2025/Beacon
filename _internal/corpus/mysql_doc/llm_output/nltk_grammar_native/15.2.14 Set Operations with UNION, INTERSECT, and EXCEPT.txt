```
S -> SET_OPERATION_STATEMENT

SET_OPERATION_STATEMENT -> QUERY_EXPRESSION
QUERY_EXPRESSION -> WITH_CLAUSE QUERY_EXPRESSION_BODY ORDER_BY_CLAUSE LIMIT_CLAUSE INTO_CLAUSE
                   | QUERY_EXPRESSION_BODY ORDER_BY_CLAUSE LIMIT_CLAUSE INTO_CLAUSE

QUERY_EXPRESSION_BODY -> QUERY_TERM
                       | QUERY_EXPRESSION_BODY UNION ALL_OPTION QUERY_TERM
                       | QUERY_EXPRESSION_BODY EXCEPT ALL_OPTION QUERY_TERM
                       
QUERY_TERM -> QUERY_PRIMARY
            | QUERY_TERM INTERSECT ALL_OPTION QUERY_PRIMARY

QUERY_PRIMARY -> QUERY_BLOCK
               | '(' QUERY_EXPRESSION_BODY ORDER_BY_CLAUSE LIMIT_CLAUSE INTO_CLAUSE ')'

QUERY_BLOCK -> QUERY_SPECIFICATION
             | TABLE_VALUE_CONSTRUCTOR
             | EXPLICIT_TABLE

ALL_OPTION -> 'ALL' | 'DISTINCT' | ''

WITH_CLAUSE -> 'with_clause_unknown' | ''
ORDER_BY_CLAUSE -> 'ORDER BY' ORDER_CRITERIA | ''
ORDER_CRITERIA -> 'order_criteria_unknown'

LIMIT_CLAUSE -> 'LIMIT' LIMIT_VALUE | ''
LIMIT_VALUE -> 'limit_value_unknown'

INTO_CLAUSE -> 'into_clause_unknown' | ''

QUERY_SPECIFICATION -> 'SELECT statement_unknown'
TABLE_VALUE_CONSTRUCTOR -> 'table_value_constructor_unknown'
EXPLICIT_TABLE -> 'TABLE' 'table_unknown'

# Examples of statements using these constructs
S -> UNION_STATEMENT | INTERSECT_STATEMENT | EXCEPT_STATEMENT

UNION_STATEMENT -> QUERY_BLOCK_LIST 'UNION' ALL_OPTION QUERY_BLOCK_LIST ORDER_BY_CLAUSE LIMIT_CLAUSE
INTERSECT_STATEMENT -> QUERY_BLOCK_LIST 'INTERSECT' ALL_OPTION QUERY_BLOCK_LIST ORDER_BY_CLAUSE LIMIT_CLAUSE
EXCEPT_STATEMENT -> QUERY_BLOCK_LIST 'EXCEPT' ALL_OPTION QUERY_BLOCK_LIST ORDER_BY_CLAUSE LIMIT_CLAUSE

QUERY_BLOCK_LIST -> QUERY_BLOCK ITEMS_OPTION
ITEMS_OPTION -> 'as_items' | ''
```
