```
S -> SAVEPOINT_STATEMENT | ROLLBACK_SAVEPOINT_STATEMENT | RELEASE_SAVEPOINT_STATEMENT

SAVEPOINT_STATEMENT -> 'SAVEPOINT' 'savepoint_identifier'

ROLLBACK_SAVEPOINT_STATEMENT -> 'ROLLBACK' WORK_OPTION 'TO' SAVEPOINT_OPTION 'savepoint_identifier'
WORK_OPTION -> 'WORK' | ''
SAVEPOINT_OPTION -> 'SAVEPOINT' | ''

RELEASE_SAVEPOINT_STATEMENT -> 'RELEASE' 'SAVEPOINT' 'savepoint_identifier'
```
