S -> CREATE_EVENT_STATEMENT

CREATE_EVENT_STATEMENT -> 'CREATE' DEFINER_OPTION 'EVENT' IF_NOT_EXISTS_OPTION 'new_event_name_1' 'ON SCHEDULE' SCHEDULE ON_COMPLETION_OPTION STATUS_OPTION COMMENT_OPTION 'DO' EVENT_BODY

DEFINER_OPTION -> 'DEFINER' '=' 'user_name_1' | ''
IF_NOT_EXISTS_OPTION -> 'IF NOT EXISTS' | ''
ON_COMPLETION_OPTION -> ON_COMPLETION 'NOT PRESERVE' | ON_COMPLETION 'PRESERVE' | ''
ON_COMPLETION -> 'ON COMPLETION'

STATUS_OPTION -> 'ENABLE' | 'DISABLE' | 'DISABLE ON REPLICA' | 'DISABLE ON SLAVE' | ''

COMMENT_OPTION -> 'COMMENT' STRING_LITERAL | ''

SCHEDULE -> AT_SCHEDULE | EVERY_SCHEDULE
AT_SCHEDULE -> 'AT' 'timestamp_value' INTERVAL_OPTION
EVERY_SCHEDULE -> 'EVERY' INTERVAL STARTS_CLAUSE ENDS_CLAUSE

INTERVAL_OPTION -> INTERVAL | ''
STARTS_CLAUSE -> 'STARTS' 'timestamp_value' INTERVAL_OPTION | ''
ENDS_CLAUSE -> 'ENDS' 'timestamp_value' INTERVAL_OPTION | ''

INTERVAL -> 'quantity_value' INTERVAL_TYPE
INTERVAL_TYPE -> 'YEAR' | 'QUARTER' | 'MONTH' | 'DAY' | 'HOUR' | 'MINUTE'
               | 'WEEK' | 'SECOND' | 'YEAR_MONTH' | 'DAY_HOUR' | 'DAY_MINUTE'
               | 'DAY_SECOND' | 'HOUR_MINUTE' | 'HOUR_SECOND' | 'MINUTE_SECOND'

EVENT_BODY -> 'event_body_unknown'

STRING_LITERAL -> "'string_value'"
