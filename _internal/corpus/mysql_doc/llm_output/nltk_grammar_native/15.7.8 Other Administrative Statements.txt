```
S -> BINLOG_STATEMENT | CACHE_INDEX_STATEMENT | FLUSH_STATEMENT 
    | LOAD_INDEX_INTO_CACHE_STATEMENT | RESET_STATEMENT 
    | RESTART_STATEMENT | SHUTDOWN_STATEMENT

BINLOG_STATEMENT -> 'BINLOG' STR
STR -> "'base64_string_value'"

CACHE_INDEX_STATEMENT -> 'CACHE' 'INDEX' INDEX_SPECIFICATION 'IN' KEY_CACHE_NAME
INDEX_SPECIFICATION -> TBL_INDEX_LIST | TBL_NAME_PARTITION
TBL_INDEX_LIST -> TBL_NAME INDEX_OPTION | TBL_NAME INDEX_OPTION ',' TBL_INDEX_LIST
TBL_NAME_PARTITION -> 'tbl_name' 'PARTITION' '(' PARTITION_LIST ')'
INDEX_OPTION -> INDEX_OPTION_TYPE '(' INDEX_NAME_LIST ')'
INDEX_OPTION_TYPE -> 'INDEX' | 'KEY'
INDEX_NAME_LIST -> 'index_name' | 'index_name' ',' INDEX_NAME_LIST
PARTITION_LIST -> PARTITION_NAME | PARTITION_NAME ',' PARTITION_LIST | 'ALL'
PARTITION_NAME -> 'partition_name'
KEY_CACHE_NAME -> 'key_cache_name'

FLUSH_STATEMENT -> 'FLUSH' NO_WRITE_OPTION FLUSH_BODY
NO_WRITE_OPTION -> 'NO_WRITE_TO_BINLOG' | 'LOCAL' | ''
FLUSH_BODY -> FLUSH_OPTIONS | TABLES_OPTIONS
FLUSH_OPTIONS -> FLUSH_OPTION | FLUSH_OPTION ',' FLUSH_OPTIONS
FLUSH_OPTION -> BINARY_LOGS | ENGINE_LOGS | ERROR_LOGS | GENERAL_LOGS 
               | LOGS | PRIVILEGES | OPTIMIZER_COSTS | RELAY_LOGS 
               | SLOW_LOGS | STATUS | USER_RESOURCES
BINARY_LOGS -> 'BINARY LOGS'
ENGINE_LOGS -> 'ENGINE LOGS'
ERROR_LOGS -> 'ERROR LOGS'
GENERAL_LOGS -> 'GENERAL LOGS'
LOGS -> 'LOGS'
PRIVILEGES -> 'PRIVILEGES'
OPTIMIZER_COSTS -> 'OPTIMIZER_COSTS'
RELAY_LOGS -> 'RELAY LOGS' CHANNEL_OPTION
CHANNEL_OPTION -> 'FOR CHANNEL' 'channel' | ''
SLOW_LOGS -> 'SLOW LOGS'
STATUS -> 'STATUS'
USER_RESOURCES -> 'USER_RESOURCES'
TABLES_OPTIONS -> TABLE_SYNONYM TABLE_LIST LOCK_OPTION
TABLE_SYNONYM -> 'TABLE' | 'TABLES'
TABLE_LIST -> 'tbl_name' | 'tbl_name' ',' TABLE_LIST | ''
LOCK_OPTION -> 'WITH READ LOCK' | 'FOR EXPORT' | ''

LOAD_INDEX_INTO_CACHE_STATEMENT -> 'LOAD INDEX INTO CACHE' LOAD_INDEX_SPEC
LOAD_INDEX_SPEC -> TBL_INDEX_LIST IGNORE_LEAVES_OPTION
IGNORE_LEAVES_OPTION -> 'IGNORE LEAVES' | ''

RESET_STATEMENT -> 'RESET' RESET_OPTIONS
RESET_OPTIONS -> RESET_OPTION | RESET_OPTION ',' RESET_OPTIONS
RESET_OPTION -> 'BINARY LOGS AND GTIDS' | 'REPLICA'

RESTART_STATEMENT -> 'RESTART'
```
