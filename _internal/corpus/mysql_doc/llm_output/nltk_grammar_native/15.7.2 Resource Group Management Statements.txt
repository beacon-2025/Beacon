ALTER_ATTRIBUTES -> VCPU_OPTION THREAD_PRIORITY_OPTION ENABLE_DISABLE_OPTION
VCPU_OPTION -> 'VCPU' '=' VCPU_SPEC | ''
VCPU_SPEC -> 'N' | 'M - N' | VCPU_SPEC ',' VCPU_SPEC  # Comma-separated list of VCPU specifiers
THREAD_PRIORITY_OPTION -> 'THREAD_PRIORITY' '=' 'int_value' | ''
ENABLE_DISABLE_OPTION -> ENABLE_OPTION | DISABLE_OPTION | ''
ENABLE_OPTION -> 'ENABLE'
DISABLE_OPTION -> 'DISABLE' FORCE_OPTION
FORCE_OPTION -> 'FORCE' | ''

GROUP_TYPE -> 'SYSTEM' | 'USER'
CREATE_ATTRIBUTES -> VCPU_OPTION THREAD_PRIORITY_OPTION ENABLE_DISABLE_OPTION
# Re-use VCPU_OPTION, THREAD_PRIORITY_OPTION, ENABLE_DISABLE_OPTION from ALTER

DROP_RESOURCE_GROUP -> 'DROP' 'RESOURCE GROUP' 'group_name_1' FORCE_OPTION
# Re-use FORCE_OPTION from ALTER

SET_RESOURCE_GROUP -> 'SET' 'RESOURCE GROUP' 'group_name_1' FOR_OPTION
FOR_OPTION -> 'FOR' THREAD_IDS | ''
THREAD_IDS -> 'thread_id_value' | THREAD_IDS ',' 'thread_id_value'
