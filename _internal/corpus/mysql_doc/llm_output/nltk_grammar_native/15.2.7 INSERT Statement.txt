```
S -> INSERT_VALUES | INSERT_SET | INSERT_SELECT

# INSERT with explicit values syntax 
INSERT_VALUES -> 'INSERT' PRIORITY IGNORE 'INTO' 'table_name_1' PARTITION_OPTION COLUMN_LIST VALUES_AS_OPTION ROW_ALIAS_DUPLICATE
VALUES_AS_OPTION -> 'VALUES' VALUE_LIST | 'VALUE' VALUE_LIST
PARTITION_OPTION -> 'PARTITION' '(' PARTITION_LIST ')' | ''
PARTITION_LIST -> 'partition_name_1' | 'partition_name_1' ',' PARTITION_LIST
COLUMN_LIST -> '(' COLUMN_NAME_LIST ')' | ''
COLUMN_NAME_LIST -> 'column_name_1' | 'column_name_1' ',' COLUMN_NAME_LIST
VALUE_LIST -> '(' VALUE_ELEMENTS ')' | '(' VALUE_ELEMENTS ')' ',' VALUE_LIST
VALUE_ELEMENTS -> VALUE | VALUE ',' VALUE_ELEMENTS
VALUE -> 'expr_value' | 'DEFAULT'
ROW_ALIAS_DUPLICATE -> AS_ROW_ALIAS ON_DUPLICATE_KEY_UPDATE | ON_DUPLICATE_KEY_UPDATE
AS_ROW_ALIAS -> 'AS' 'row_alias_unknown' AS_COLUMN_ALIAS_LIST | ''
AS_COLUMN_ALIAS_LIST -> '(' AS_COLUMN_NAME_LIST ')' | ''
AS_COLUMN_NAME_LIST -> 'col_alias_unknown' | 'col_alias_unknown' ',' AS_COLUMN_NAME_LIST
ON_DUPLICATE_KEY_UPDATE -> 'ON DUPLICATE KEY UPDATE' ASSIGNMENT_LIST | ''
ASSIGNMENT_LIST -> ASSIGNMENT | ASSIGNMENT ',' ASSIGNMENT_LIST
ASSIGNMENT -> 'column_name_2' '=' VALUE | 'row_alias_unknown.column_name_2' | 'table_name_1.column_name_2' | 'row_alias_unknown.col_alias_unknown'

# INSERT with SET syntax 
INSERT_SET -> 'INSERT' PRIORITY IGNORE 'INTO' 'table_name_2' PARTITION_OPTION SET_CLAUSE ROW_ALIAS_DUPLICATE
SET_CLAUSE -> 'SET' ASSIGNMENT_LIST

# INSERT with SELECT syntax (also supports "TABLE" and "VALUES ROW()" formats)
INSERT_SELECT -> 'INSERT' PRIORITY IGNORE 'INTO' 'table_name_3' PARTITION_OPTION COLUMN_LIST SELECT_VALUES_TABLE ON_DUPLICATE_KEY_UPDATE
SELECT_VALUES_TABLE -> SELECT_STATEMENT | 'TABLE' 'table_name_4' | 'VALUES' ROW_CONSTRUCTOR_LIST
ROW_CONSTRUCTOR_LIST -> 'ROW' '(' VALUE_ELEMENTS ')' | 'ROW' '(' VALUE_ELEMENTS ')' ',' ROW_CONSTRUCTOR_LIST
SELECT_STATEMENT -> 'SELECT' 'select_unknown'
PRIORITY -> 'LOW_PRIORITY' | 'DELAYED' | 'HIGH_PRIORITY' | ''
IGNORE -> 'IGNORE' | ''
```
