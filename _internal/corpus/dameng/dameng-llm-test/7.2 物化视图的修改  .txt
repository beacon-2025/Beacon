-- 创建模式
CREATE SCHEMA PURCHASING;
-- 创建物化视图
CREATE MATERIALIZED VIEW PURCHASING.MV_VENDOR_EXCELLENT AS
SELECT * FROM DUAL;
-- 创建另一个物化视图

-- 用例7: 修改物化视图MV_VENDOR_EXCELLENT为启用查询改写
